```python
import aiohttp
from typing import List, Dict, Optional
import asyncio

BASE_URL = "https://gorzdrav.spb.ru/_api/api/v2"
HEADERS = {"User-Agent": "MedBot/1.0 (Telegram; +79000000000)"}

async def _safe_request(url: str) -> List[Dict]:
    """Универсальная функция с обработкой разных форматов ответа"""
    try:
        timeout = aiohttp.ClientTimeout(total=15)
        async with aiohttp.ClientSession(timeout=timeout) as session:
            async with session.get(url, headers=HEADERS) as resp:
                if resp.status != 200:
                    return []
                data = await resp.json()
                
                # Пробуем разные возможные форматы ответа
                if isinstance(data, list):
                    return data
                if isinstance(data, dict):
                    if "result" in data:
                        return data["result"] if isinstance(data["result"], list) else []
                    if "data" in data:
                        return data["data"] if isinstance(data["data"], list) else []
                return []
    except Exception as e:
        print(f"API error for {url}: {e}")
        return []

async def get_districts() -> List[Dict]:
    return await _safe_request(f"{BASE_URL}/shared/districts")

async def get_lpus_by_district(district_id: int) -> List[Dict]:
    return await _safe_request(f"{BASE_URL}/shared/district/{district_id}/lpus")

async def get_specialties(lpu_id: int) -> List[Dict]:
    return await _safe_request(f"{BASE_URL}/schedule/lpu/{lpu_id}/specialties")

async def get_doctors(lpu_id: int, specialty_id: int) -> List[Dict]:
    return await _safe_request(f"{BASE_URL}/schedule/lpu/{lpu_id}/speciality/{specialty_id}/doctors")

async def get_free_appointments(lpu_id: int, doctor_id: int) -> List[Dict]:
    return await _safe_request(f"{BASE_URL}/schedule/lpu/{lpu_id}/doctor/{doctor_id}/appointments")
```

```python
import aiohttp

async def call_qwen(prompt: str, retry: int = 1) -> str:
    """Вызов Qwen с fallback и таймаутом"""
    try:
        timeout = aiohttp.ClientTimeout(total=20)
        async with aiohttp.ClientSession(timeout=timeout) as session:
            async with session.post(
                "https://openrouter.ai/api/v1/chat/completions",
                headers={
                    "Authorization": f"Bearer {OPENROUTER_API_KEY}",
                    "Content-Type": "application/json"
                },
                json={
                    "model": "qwen/qwen3.6-plus-preview",
                    "messages": [{"role": "user", "content": prompt}],
                    "temperature": 0.3,
                    "max_tokens": 1500
                }
            ) as resp:
                if resp.status != 200:
                    error_text = await resp.text()
                    if retry > 0:
                        await asyncio.sleep(1)
                        return await call_qwen(prompt, retry - 1)
                    return f"⚠️ Ошибка API: {resp.status}\nПопробуйте позже."
                
                data = await resp.json()
                return data["choices"][0]["message"]["content"] + DISCLAIMER
    except asyncio.TimeoutError:
        return "⏰ Таймаут API. Попробуйте позже." + DISCLAIMER
```

5. Важные технические моменты

· Вебхук или отдельный сервер: Авторизация через OAuth 2.0 требует наличия у бота публичного URL (вебхука) для приема callback-запросов от Госуслуг, так как просто Telegram бот не может напрямую перехватить параметры URL.
· Регистрация приложения: Вам потребуется зарегистрировать приложение в личном кабинете разработчика ЕСИА и получить client_id и client_secret.
· Ключи шифрования: Для подписи JWT-токенов при обмене кода на токен обычно требуется RSA-ключ.
